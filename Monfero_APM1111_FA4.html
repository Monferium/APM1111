Summary of Formulas:¶

Moment Degree Raw Moment (Origin = 0) Central Moment (Mean)
1st Moment (Arithmetic Mean) $ m'_1 = \frac{\sum_{j=1}^{N}X_{j}}{N} $ $ m_1 = \frac{\sum_{j=1}^{N}(X_{j} - \mu)}{N} = 0 $
2nd Moment (Variance) $ m'_2 = \frac{\sum_{j=1}^{N}(X_{j})^2}{N} $ $ m_2 = \frac{\sum_{j=1}^{N}(X_{j} - \mu)^2}{N} = \sigma^2 $
3rd Moment (Skewness) $ m'_3 = \frac{\sum_{j=1}^{N}(X_{j})^3}{N} $ $ m_3 = \frac{\sum_{j=1}^{N}(X_{j} - \mu)^3}{N} $
4th Moment (Kurtosis) $ m'_4 = \frac{\sum_{j=1}^{N}(X_{j})^4}{N} $ $ m_4 = \frac{\sum_{j=1}^{N}(X_{j} - \mu)^4}{N} $

# Create a table to store results
results_A = {
    'Moments of Degree': [1, 2, 3, 4],
    'Normal': [],
    'Skewed-right': [],
    'Skewed-left': [],
    'Uniform': []
}

# Calculate raw moments (Origin = 0) for each series of dataset df
for moment_degree in range(1,5):
    for each_series in df.columns:
        moment_summation = 0
        for current_value in df[each_series]:
            moment_summation += current_value ** moment_degree
        result_moment = round(moment_summation/len(df[each_series]), 2)
        results_A[each_series].append(result_moment)

# Create a table to store results
results_B = {
    'Central Moments of Degree': [1, 2, 3, 4],
    'Normal': [],
    'Skewed-right': [],
    'Skewed-left': [],
    'Uniform': []
}

# Calculate raw moments (Origin = 0) for each series of dataset df
for moment_degree in range(1,5):
    for each_series in df.columns:
        central_summation = 0
        each_mean = sum(df[each_series]) / len(df[each_series])
        for current_value in df[each_series]:
            central_summation += (current_value - each_mean) ** moment_degree
        result_moment = round(central_summation/len(df[each_series]), 2)
        results_B[each_series].append(result_moment)

moment_results_B = pd.DataFrame(results_B)
moment_results_B.rename_axis('Task B Result ID')
